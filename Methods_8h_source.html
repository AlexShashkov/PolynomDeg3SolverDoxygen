102 void KahanQuadratic(number a, number b, number c, vector<complex<number>> &roots){
103 // x^2, x, c
104 //Coefficients should be in CBA order
105 try{
106 b = b/static_cast<number>(-2);
107 if(!std::isfinite(a)) throw std::invalid_argument("Коэффициент при x^2 равен нулю или б/м.");
108 if(!std::isfinite(b)) throw std::invalid_argument("Коэффициент при x равен нулю или б/м.");
109 number p = b*b;
110 number q = a*c;
111 //Use the hardware's FMA
112 number dp = fma(b,b,-p);
113 number dq = fma(a,c,-q);
114 // дискриминант
115 number d = (p-q) + (dp - dq);
116 d = std::max(d,static_cast<number>(0));
117 number S = b;
118 S = std::sqrt(d)*(sign(S) + (S==0)) + S;
119 number Z1 = S/a;
120 number Z2 = c/S;
121 if(anynotfinite(Z1, Z2)) throw std::invalid_argument("Полученные корни не определены.");
122
123 roots = {Z1, Z2};
124 }

126 std::cerr << "Error occured while working with " << a << " " << b << " " << c << " " << "\n";
